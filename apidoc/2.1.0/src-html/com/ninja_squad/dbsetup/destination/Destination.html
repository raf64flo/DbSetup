025package com.ninja_squad.dbsetup.destination;
026
027import java.sql.Connection;
028import java.sql.SQLException;
029
030import com.ninja_squad.dbsetup.DbSetupTracker;
031
032/**
033 * The destination of a database setup. It's advised to make implementations of this
034 * interface immutable, and to make them implement equals and hashCode in order for {@link DbSetupTracker} to function
035 * properly, or to make them singletons.
036 * @author JB Nizet
037 */
038public interface Destination {
039    /**
040     * Returns a connection to the destination database
041     * @return a connection to the destination database
042     * @throws SQLException if a connection can't be returned
043     */
044    Connection getConnection() throws SQLException;
045}
